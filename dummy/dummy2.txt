### âœ… 1. `auth.js`: Utility to handle auth

```js
// src/auth.js
export const setToken = (token) => {
  localStorage.setItem('token', token);
};

export const getToken = () => {
  return localStorage.getItem('token');
};

export const isLoggedIn = () => {
  return !!getToken();
};

export const logout = () => {
  localStorage.removeItem('token');
};
```

### âœ… 2. `Login.js`: Simple login form

```jsx
// src/pages/Login.js
import React, { useState } from 'react';
import { setToken } from '../auth';
import { useNavigate } from 'react-router-dom';

const Login = () => {
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const navigate = useNavigate();

  const handleLogin = async (e) => {
    e.preventDefault();
    try {
      const res = await fetch('http://localhost:5000/api/login', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ username, password })
      });

      if (!res.ok) throw new Error('Login failed');
      const data = await res.json();
      setToken(data.token); // Save token in localStorage
      navigate('/upload');
    } catch (err) {
      alert(err.message);
    }
  };

  return (
    <form onSubmit={handleLogin}>
      <h2>Login</h2>
      <input placeholder="Username" value={username} onChange={(e) => setUsername(e.target.value)} />
      <input type="password" placeholder="Password" value={password} onChange={(e) => setPassword(e.target.value)} />
      <button type="submit">Login</button>
    </form>
  );
};

export default Login;
```

## âœ… React Upload Component (with Auth Token)

### ðŸ“„ `Upload.js` (Updated Protected Upload Page)

```jsx
import React, { useState, useEffect } from 'react';
import { useNavigate } from 'react-router-dom';
import { getToken, isLoggedIn } from '../auth';
import axios from 'axios';

const Upload = () => {
  const [file, setFile] = useState(null);
  const [type, setType] = useState('image'); // 'image' or 'doc'
  const [uploadUrl, setUploadUrl] = useState('');
  const navigate = useNavigate();

  useEffect(() => {
    if (!isLoggedIn()) {
      navigate('/login');
    }
  }, [navigate]);

  const handleUpload = async () => {
    if (!file) {
      alert('Please select a file');
      return;
    }

    const formData = new FormData();
    formData.append(type === 'image' ? 'image' : 'doc', file);

    const url =
      type === 'image'
        ? 'http://localhost:5000/images/upload'
        : 'http://localhost:5000/docs/docupload';

    try {
      const res = await axios.post(url, formData, {
        headers: {
          'Content-Type': 'multipart/form-data',
          Authorization: `Bearer ${getToken()}`
        }
      });

      setUploadUrl(res.data.url);
      alert('Upload successful!');
    } catch (err) {
      console.error(err);
      alert(err.response?.data || 'Upload failed');
    }
  };

  return (
    <div>
      <h2>Upload Page (Protected)</h2>

      <label>
        Select Type:
        <select value={type} onChange={(e) => setType(e.target.value)}>
          <option value="image">Image</option>
          <option value="doc">Document</option>
        </select>
      </label>

      <br />

      <input type="file" onChange={(e) => setFile(e.target.files[0])} />

      <button onClick={handleUpload}>Upload</button>

      {uploadUrl && (
        <div>
          <p>Uploaded File URL:</p>
          <a href={uploadUrl} target="_blank" rel="noreferrer">{uploadUrl}</a>
        </div>
      )}
    </div>
  );
};

export default Upload;
```

## âœ… Backend: Add Auth Middleware for Upload Routes

If you're securing your upload routes, you need to **verify the JWT** on the backend.

### ðŸ“„ `middleware/auth.js`

```js
const jwt = require('jsonwebtoken');
const SECRET = 'your-secret-key'; // Use the same key as in your login route

const verifyToken = (req, res, next) => {
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1]; // Bearer <token>

  if (!token) return res.status(401).send('Access denied. No token provided.');

  try {
    const decoded = jwt.verify(token, SECRET);
    req.user = decoded;
    next();
  } catch (err) {
    res.status(403).send('Invalid token');
  }
};

module.exports = verifyToken;
```
